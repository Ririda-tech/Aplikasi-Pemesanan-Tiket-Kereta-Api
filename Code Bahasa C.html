1 #include <stdio.h>
  2  #include <string.h>
  3  #include <stdlib.h>
  4  #include <windows.h> // Untuk Sleep function di Windows
  5  #define TARIF_KERETA_PER_KM 1000
  6  #define MAKS_RUTE 5
  7  #define MAKS_JADWAL 3
  8  typedef struct {
  9      char rute[50];  // Nama rute
 10      int jarak;      // Jarak (km)
 11      int biaya;      // Biaya perjalanan
 12      char jadwal[MAKS_JADWAL][10]; // Jadwal keberangkatan
 13  } Kereta;
 14  
 15  // Prototipe fungsi
 16  void tampilkanRuteDenganJadwal(Kereta rute[], int jumlah);
 17  int pilihRute(Kereta rute[], int jumlah, int *pilihJadwal);
 18  void cetakStruk(Kereta rute, char jadwal[], int jumlahTiket);
 19  void prosesPembayaran(int totalBiaya);
 20  void opsiSetelahPembayaran(int totalBiaya);
 21  void hubungiCustomerService();
 22  void clearScreen(int seconds); // Fungsi untuk membersihkan layar setelah beberapa detik
 23  void setColor(int color); // Fungsi untuk mengatur warna teks
 24  
 25  int main() {
 26      Kereta rute[MAKS_RUTE] = {
 27          {"Jakarta - Bandung", 150, 0, {"08:00", "12:00", "16:00"}},
 28          {"Bandung - Surabaya", 600, 0, {"07:00", "13:00", "19:00"}},
 29          {"Surabaya - Yogyakarta", 400, 0, {"06:30", "11:30", "18:30"}},
 30          {"Jakarta - Semarang", 300, 0, {"09:00", "14:00", "20:00"}},
 31          {"Semarang - Solo", 120, 0, {"05:00", "10:00", "15:00"}}
 32      };
 33  
 34      // Hitung biaya perjalanan untuk setiap rute
 35      for (int i = 0; i < MAKS_RUTE; i++) {
 36          rute[i].biaya = rute[i].jarak * TARIF_KERETA_PER_KM;
 37      }
 38  
 39      int pilihan, jumlahTiket, pilihJadwal;
 40  
 41      do {
 42          setColor(10); // Mengatur warna teks hijau
 43          printf("\n----- Aplikasi Pemesanan Tiket Kereta ------\n");
 44          setColor(15); // Mengatur warna teks putih
 45          printf("1. Tampilkan daftar rute dan jadwal\n");
 46          printf("2. Pesan tiket kereta\n");
 47          printf("0. Keluar\n");
 48          printf("Pilih menu: ");
 49          scanf("%d", &pilihan);
 50  
 51          clearScreen(1); // Bersihkan layar setelah memilih menu
 52  
 53          switch (pilihan) {
 54              case 1:
 55                  tampilkanRuteDenganJadwal(rute, MAKS_RUTE);
 56                  break;
 57              case 2: {
 58                  int indeks = pilihRute(rute, MAKS_RUTE, &pilihJadwal);
 59                  if (indeks != -1) {
 60                      printf("Masukkan jumlah tiket: ");
 61                      scanf("%d", &jumlahTiket);
 62                      cetakStruk(rute[indeks], rute[indeks].jadwal[pilihJadwal], jumlahTiket);
 63                      prosesPembayaran(rute[indeks].biaya * jumlahTiket);
 64                  }
 65                  break;
 66              }
 67              case 0:
 68                  setColor(12); // Mengatur warna teks merah untuk keluar
 69                  printf("Keluar dari program. Terima kasih!\n");
 70                  setColor(15); // Kembalikan ke warna default
 71                  break;
 72              default:
 73                  printf("Pilihan tidak valid. Coba lagi.\n");
 74          }
 75      } while (pilihan != 0);
 76  
 77      return 0;
 78  }

 80  // Fungsi untuk menampilkan daftar rute beserta jadwalnya
 81  void tampilkanRuteDenganJadwal(Kereta rute[], int jumlah) {
 82      setColor(11); // Mengatur warna teks biru terang
 83      printf("\n------------ DAFTAR RUTE KERETA ------------\n");
 84      setColor(15); // Mengatur warna teks putih
 85      for (int i = 0; i < jumlah; i++) {
 86          printf("%d. %s\n", i + 1, rute[i].rute);
 87          printf("   o Jarak: %d km\n", rute[i].jarak);
 88          printf("   o Biaya: Rp %d\n", rute[i].biaya);
 89          printf("   o Jadwal: ");
 90          for (int j = 0; j < MAKS_JADWAL; j++) {
 91              printf("%s ", rute[i].jadwal[j]);
 92          }
 93          printf("\n");
 94      }
 95      printf("--------------------------------------------\n");
 96  }
 97  
 98  // Fungsi untuk memilih rute dan jadwal
 99  int pilihRute(Kereta rute[], int jumlah, int *pilihJadwal) {
100      int pilihan;
101      setColor(10); // Mengatur warna teks hijau
102      tampilkanRuteDenganJadwal(rute, jumlah);
103      setColor(15); // Mengatur warna teks putih
104      printf("Pilih rute (1-%d): ", jumlah);
105      scanf("%d", &pilihan);
106  
107      if (pilihan >= 1 && pilihan <= jumlah) {
108          printf("Pilih jadwal keberangkatan (1-%d): ", MAKS_JADWAL);
109          scanf("%d", pilihJadwal);
110          if (*pilihJadwal >= 1 && *pilihJadwal <= MAKS_JADWAL) {
111              return pilihan - 1; // Indeks array
112          } else {
113              setColor(12); // Mengatur warna teks merah untuk error
114              printf("Jadwal tidak valid.\n");
115              setColor(15); // Kembalikan ke warna default
116              return -1;
117          }
118      } else {
119          setColor(12); // Mengatur warna teks merah untuk error
120          printf("Rute tidak valid.\n");
121          setColor(15); // Kembalikan ke warna default
122          return -1;
123      }
124  }

126  // Fungsi untuk mencetak struk tiket
127  void cetakStruk(Kereta rute, char jadwal[], int jumlahTiket) {
128      int totalBiaya = rute.biaya * jumlahTiket;
129      setColor(14); // Mengatur warna teks kuning
130      printf("\n------------ STRUK TIKET KERETA ------------\n");
131      setColor(15); // Mengatur warna teks putih
132      printf("Rute           : %s\n", rute.rute);
133      printf("Jadwal         : %s\n", jadwal);
134      printf("Jarak          : %d km\n", rute.jarak);
135      printf("Biaya per tiket: Rp %d\n", rute.biaya);
136      printf("Jumlah tiket   : %d\n", jumlahTiket);
137      printf("Total biaya    : Rp %d\n", totalBiaya);
138      printf("--------------------------------------------\n");
139  }

141  // Fungsi untuk proses pembayaran
142  void prosesPembayaran(int totalBiaya) {
143      int metode;
144      do {
145          setColor(10); // Mengatur warna teks hijau
146          printf("\n--- Proses Pembayaran ---\n");
147          setColor(15); // Mengatur warna teks putih
148          printf("Total yang harus dibayar: Rp %d\n", totalBiaya);
149          printf("Pilih metode pembayaran:\n");
150          printf("1. Transfer Bank\n");
151          printf("2. Virtual Account\n");
152          printf("3. Hubungi Customer Service untuk bantuan\n");
153          printf("Pilih metode (1-3): ");
154          scanf("%d", &metode);
155  
156          switch (metode) {
157              case 1:
158                  setColor(10); // Mengatur warna teks hijau
159                  printf("Silakan transfer ke rekening berikut:\n");
160                  printf("Bank: BNI\n");
161                  printf("Nomor Rekening: 1234567890\n");
162                  break;
163              case 2:
164                  setColor(10); // Mengatur warna teks hijau
165                  printf("Virtual Account:\n");
166                  printf("Bank Mandiri: 9876543210\n");
167                  break;
168              case 3:
169                  hubungiCustomerService();
170                  break;
171              default:
172                  setColor(12); // Mengatur warna teks merah untuk error
173                  printf("Metode tidak valid. Coba lagi.\n");
174                  setColor(15); // Kembalikan ke warna default
175          }
176      } while (metode < 1 || metode > 3);
177      opsiSetelahPembayaran(totalBiaya);
178  }

180  // Fungsi untuk opsi setelah pembayaran
181  void opsiSetelahPembayaran(int totalBiaya) {
182      int opsi;
183      do {
184          setColor(10); // Mengatur warna teks hijau
185          printf("\n--- Opsi Setelah Pembayaran ---\n");
186          setColor(15); // Mengatur warna teks putih
187          printf("1. Ganti Metode Pembayaran\n");
188          printf("2. Hubungi Customer Service\n");
189          printf("3. Keluar dari Program dengan Pembatalan Pemesanan Tiket\n");
190          printf("Pilih opsi (1-3): ");
191          scanf("%d", &opsi);
192  
193          switch (opsi) {
194              case 1:
195                  prosesPembayaran(totalBiaya);
196                  break;
197              case 2:
198                  hubungiCustomerService();
199                  break;
200              case 3:
201                  setColor(12); // Mengatur warna teks merah untuk keluar
202                  printf("\nPembatalan Pemesanan Tiket...\n");
203                  Sleep(2); // Tunggu beberapa detik untuk tampilan pembatalan
204                  setColor(15); // Kembalikan ke warna default
205                  printf("Pesanan tiket dibatalkan. Keluar dari program.\n");
206                  exit(0); // Keluar dari program
207                  break;
208              default:
209                  setColor(12); // Mengatur warna teks merah untuk error
210                  printf("Opsi tidak valid. Coba lagi.\n");
211                  setColor(15); // Kembalikan ke warna default
212          }
213      } while (opsi != 3); // Akan keluar dari loop dan program jika opsi 3 dipilih
214  }
